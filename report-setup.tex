\usepackage[english]{babel}


% LAYOUT
\setlength\parindent{0in}
\setlength{\parskip}{1em}

% TYPOGRAPHY
% font encoding
\usepackage[T1]{fontenc}
% main font: linux libertine
\usepackage{libertine}
% monospace font: Bitstream Vera
\setmonofont[Scale=0.83]{Bitstream Vera Sans Mono}

% Disable Section Numbering
\setcounter{secnumdepth}{-1}

% Eliminate itemize spacing
\usepackage{enumitem}
\setlist{nolistsep}

% Math
\usepackage{amsmath,amsfonts,amsthm}

% Table
\usepackage{booktabs}
\usepackage{tabulary}

% Color
% Add more options to xcolor package. The package itself will be loaded
% by minted package.
\PassOptionsToPackage{dvipsnames}{xcolor}

% Code/Source Listings
\usepackage{minted}

% GRAPHICX
\usepackage{graphicx}
\usepackage{subcaption}

% Hyperref
\usepackage{hyperref}
\hypersetup{,
    pdfcreator={XelaTeX},    % creator of the document
    colorlinks=true,         % colored links
    linkcolor=BrickRed,      % color of internal links
    citecolor=OliveGreen,    % color of links to bibliography
    filecolor=magenta,       % color of file links
    urlcolor=BlueViolet      % color of external links
}
